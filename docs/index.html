üöÄ Quick Start

import { create } from "@omniqueue/core";
import "@omniqueue/rabbitmq";

const mq = await create("rabbitmq", { url: "amqp://localhost" });
await mq.send("tasks", { body: { id: 1 } });
await mq.receive("tasks", async msg => {
  console.log("got", msg.body);
  await msg.ack();
});
